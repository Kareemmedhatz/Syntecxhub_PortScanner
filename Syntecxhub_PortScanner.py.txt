import socket 

for port in range(1, 1000):
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    result = sock.connect_ex(("142.250.74.174", port))

    if result == 0:
        print("Port " + str(port) + " is open")
    else:
        print("Port " + str(port) + " is close")
    sock.close()
